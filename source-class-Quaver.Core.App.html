119     /**
120      * Run instance.
121      */
122     public function run()
123     {
124         $config = Config::getInstance();
125         
126         // Start db
127         $db = DB::getInstance();
128         $db->setConnection();
129 
130         // Load default routes
131         if (!$this->router->routes) {
132             $this->addPath('/', GLOBAL_PATH.'/../Routes.yml');
133         }
134 
135         // Manage folders and plugins
136         if (!defined('AJAX_METHOD')) {
137             // Create if not exist
138             $this->checkFiles($config->params->folders);
139 
140             // Start plugins
141             $config->setPluginsYML($this->router->modules);
142 
143             // Load routes of module
144             foreach ($config->plugins as $key => $plugin) {
145                 if ($plugin['enabled']) {
146                     if ($plugin['params']['useRoutes']) {
147                         $this->router->addPath('/', VENDOR_PATH.'/'.$plugin['packageName'].'/'.$plugin['namespacePath'].'/'.'Routes.yml', true);
148                     }
149                 }
150             }
151         }
152         
153         // Load language
154         $GLOBALS['_lang'] = new Lang();
155         if (isset($_GET['lang'])) {
156             $lang_slug = substr($_GET['lang'], 0, 3);
157             $GLOBALS['_lang']->getFromSlug($lang_slug, false, $config->params->core['defaultLang']);
158             $GLOBALS['_lang']->setCookie();
159         } else {
160             $GLOBALS['_lang']->getSiteLanguage($config->params->core['defaultLang'], $config->params->core['forcedLang']);
161         }
162 
163         // Load user
164         $GLOBALS['_user'] = new User();
165         if (isset($_COOKIE[$config->cookies['cookieName'].'_log'])) {
166             $GLOBALS['_user']->getFromCookie($_COOKIE[$config->cookies['cookieName'].'_log']);
167         }
168 
169         // Session
170         session_start();
171         if (isset($_SESSION['logged'])) {
172             if ($_SESSION['logged'] == true && $_SESSION['uID'] != 0) {
173                 $GLOBALS['_user']->getFromId($_SESSION['uID']);
174             }
175         }
176 
177         // Routing
178         if (!defined('AJAX_METHOD')) {
179             if ((!$GLOBALS['_user']->logged || $GLOBALS['_user']->logged && !$GLOBALS['_user']->isAdmin())
180                 && ($config->params->core['maintenance'])
181             ) {
182                 // Maintenance mode
183                 $this->router->dispatch('maintenance');
184             } else {
185                 // Start routing
186                 $this->router->route();
187             }
188         }
189     }

191     /**
192      * Check if exists some folders

194     private function checkFiles($folders)
195     {
196         foreach ($folders as $folder) {
197             if (!file_exists(GLOBAL_PATH.'/../'.$folder.'/')) {
198                 mkdir(GLOBAL_PATH.'/../'.$folder.'/', 0777, true);
199             }
200         }
